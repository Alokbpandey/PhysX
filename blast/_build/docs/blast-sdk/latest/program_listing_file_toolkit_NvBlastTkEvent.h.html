#ifndef NVBLASTTKEVENT_H
#define NVBLASTTKEVENT_H

#include <vector>

#include "NvBlastTypes.h"


namespace Nv
{
namespace Blast
{

// Forward declarations
class TkObject;
class TkActor;
class TkFamily;
class TkJoint;


struct TkActorData
{
    TkFamily*   family;
    void*       userData;
    uint32_t    index;
};


struct TkEvent
{
    // Enums
    enum Type
    {
        Split,
        FractureCommand,
        FractureEvent,
        JointUpdate,

        TypeCount
    };

    // Data
    const void* payload;
    Type        type;

    template<typename T>
    const T* getPayload() const { return reinterpret_cast<const T*>(payload); }
};


struct TkSplitEvent
{
    enum { EVENT_TYPE = TkEvent::Split };

    TkActorData parentData;
    uint32_t    numChildren;
    TkActor**   children;
};


struct TkFractureCommands
{
    enum { EVENT_TYPE = TkEvent::FractureCommand };

    TkActorData             tkActorData;
    NvBlastFractureBuffers  buffers;
};


struct TkFractureEvents
{
    enum { EVENT_TYPE = TkEvent::FractureEvent };

    TkActorData             tkActorData;
    NvBlastFractureBuffers  buffers;
    uint32_t                bondsDamaged;
    uint32_t                bondsBroken;
    uint32_t                chunksDamaged;
    uint32_t                chunksBroken;
};


struct TkJointUpdateEvent
{
    enum { EVENT_TYPE = TkEvent::JointUpdate };

    enum Subtype
    {
        External,
        Changed,
        Unreferenced
    };

    TkJoint*    joint;
    Subtype     subtype;
};


class TkEventListener
{
public:
    virtual void    receive(const TkEvent* events, uint32_t eventCount) = 0;
};

} // namespace Blast
} // namespace Nv


#endif // ifndef NVBLASTTKEVENT_H
